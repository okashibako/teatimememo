  certificateHandlerFile(e){
    const that = this;
    var files = [];
    files = e;
    let bundle_arr = [];
    for (var i = 0; i < files.length; i++) {
      var promise = that.fileReader(files[i]);
        promise.then(function (result) {
        console.log(result)
        bundle_arr.push(result);
        that.setState({ image: bundle_arr });
      })
    }
    that.setState({ isShowCertificate1: false });
  }
